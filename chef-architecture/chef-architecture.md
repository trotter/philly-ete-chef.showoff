# A Simple Recipe #

    @@@ruby
    # cookbooks/profile/recipes/default.rb

    home = ENV['HOME']
    git "#{home}/.profile.d" do
      # node[:profile][:repo] is set in config
      # "git://github.com/matschaffer/profile.git"
      repository node[:profile][:repo]
      action :sync
    end
    
    link "#{home}/.profile" do
      to "#{home}/.profile.d/init"
    end
